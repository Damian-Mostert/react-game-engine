			action = "jump";

		action = "run";

		updatedFalling = true;
		action = "fall" + (updatedIsLeft ? "-left" : "");
	} else {
		updatedSliding = false;
		updatedFalling = false;
	}

		position: {
			top: Math.round(updatedPosition.top),
			left: Math.round(updatedPosition.left)

		action,
		isFalling: updatedFalling,
		isAttacking: updatedAttacking,
		isSliding: updatedSliding,
